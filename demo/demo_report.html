1Written & proven in Rocq

The source — a first-order EffIR term:

Definition demo_prog : tm :=
  Bind (Perform OAsk [])
       (Bind (Perform OTrace [VVar 0])
             (Bind (Repeat 3 (incr_at key0))
                   (Perform OPut [VBytes key9; VVar 2]))).

The machine-checked theorem (full functional result for the concrete run):

Theorem demo_correct :
  let '(_, w) := run [] demo_prog (init_world (DInt 99) 0) in

2Code-generated to idiomatic OCaml 5

Direct style — the monad is erased; no interpreter, just effect calls and a native for-loop:

let demo_prog () = (let v0 = (Env.ask ()) in
  (let v1 = (Trace.emit v0) in
  (let v2 = (for _i = 1 to 3 do ignore ((let v2 = (Kv.get (Bytes.of_string "0")) in
  (let _s3 = v2 in (match _s3 with Rval.None -> (Kv.put (Bytes.of_string "0") (match (Rval.Int Z.zero) with Rval.Int _z -> Rval.Int (Z.succ _z) | _ -> raise Rval.Stuck)) | _ -> (match _s3 with Rval.Some v3 -> (Kv.put (Bytes.of_string "0") (match v3 with Rval.Int _z -> Rval.Int (Z.succ _z) | _ -> raise Rval.Stuck)) | _ -> (Kv.put (Bytes.of_string "0") (match (Rval.Int Z.zero) with Rval.Int _z -> Rval.Int (Z.succ _z) | _ -> raise Rval.Stuck))))))) done) in
  (Kv.put (Bytes.of_string "9") v0))))

Where are the effects? Env.ask / Trace.emit / Kv.put are the effect operations: each is a thin wrapper whose body is an Effect.perform, deliberately confined to runtime/ (a CI gate forbids Effect.perform anywhere else, and the .mlis hide the effect constructors). Generated code therefore reads like ordinary OCaml while the effect boundary stays narrow and reviewed:

(* runtime/env.ml *)   let ask () = Effect.perform Ask
(* runtime/trace.ml *) let emit v = Effect.perform (Emit v)
(* runtime/kv.ml *)    let put k v = Effect.perform (Put (k, v))
(* ...interpreted by Effect.Deep handlers, e.g. runtime/kv.ml: *)
(*   | effect Put (k, v), kont -> T.replace table k v; continue kont () *)
